	public static Node reverseList(Node head){
		if(head==null||head.next==null)return head;
		Node pre=null;
		Node nex=null;
		while(head!=null){
			nex=head.next;
			head.next=pre;
			pre=head;
			head=nex;
		}
		return pre;
	}

	public static Node mergeSortedList(Node head1,Node head2){
		if(head1==null)return head2;
		if(head2==null)return head1;
		Node target=null;
		if(head1.value>head2.value){

			target=head1;

		target.next=null;
		Node mergeHead=target;
		while(head1!=null && head2!=null){
			if(head1.value>head2.value){
				target.next=head2;
				head2=head2.next;
			}
			else{
				target.next=head1;

			target.next=null;
		}

				head1=head1.next;

			for(int i=0;i<len2-len1;i++){
				head2=head2.next;
			}
		}
		while(head1!=null&&head2!=null){
			if(head1==head2){
				target=head1;

			}

				head1=head1.next;

			}
		}
		return target;
	}
